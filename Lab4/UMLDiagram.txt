@startuml
skinparam classAttributeIconSize 0

class Person {
  -firstName : String
  -lastName : String
  -telNumber : String

  +getFirstName() : String
  +getLastName() : String
  +getTelNumber() : String
}

class Client {
  -billsToPay : Bill[]

  -getFreeElement(bills : billsToPay[]) : int
  +createTask(floorsNumber : int, sizeM2 : int) : Task
  +addBill(bill : Bill) : void
  +payBill(bill : Bill) : void
}

class Manager {
  ~registerTask(task : Task, bill : Bill) : Order
}

class Designer {
  ~estimatePrice(task : Task, orderNumber : int) : Bill
  ~assignBill(order : Order) : void
  ~assessOrder(order : Order) : BuilderTeam
}

class BuildingCompany {
  -name : String
  -orders : Order[]
  -manager : Manager
  -designer : Designer

  -getFreeElement(orders : Order[]) : int

  +addOrder(task : Task) : Bill
  +completeOrder(bill : Bill) : Building

  +getName() : String
  +setName(name : String) : void
}

class Order {
  +{final} task : Task
  +{final} bill : Bill
}

class BuilderTeam {
  -members : Person[]

  ~createProject(order : Order) : Building
  ~build(b : Building) : void
}

class Task {
  +{final} client : Client
  +{final} floorsNumber : int
  +{final} sizeM2 : int
}

class Building {
  +{final} floorsNumber : int
  +{final} sizeM2 : int
  -isBuilt : boolean

  +getIsBuilt() : boolean
  ~build() : void
}

class Bill {
  +{final} sum : double
  +{final} orderNumber : int
  -isPaid : boolean

  +getIsPaid() : boolean
  +pay() : void
}
